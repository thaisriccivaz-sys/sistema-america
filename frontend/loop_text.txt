                        span.style.display = has ? 'inline-block' : 'none';

                        // Gerenciar container Assinafy integrado
                        const containerAssinafy = span.nextElementSibling;
                        if (containerAssinafy && containerAssinafy.classList.contains('assinafy-integrated-container')) {
                            containerAssinafy.style.display = has ? 'flex' : 'none';
                            const btnA = containerAssinafy.querySelector('.btn-assinafy');
                            const badgeA = containerAssinafy.querySelector('.assinafy-status-badge');
                            
                            // Adicionar botÃ£o WhatsApp se nÃ£o existir
                            let btnW = containerAssinafy.querySelector('.btn-whatsapp-integrated');
                            
                            const status = docRecord?.assinafy_status || 'Nenhum';
                            const url = docRecord?.assinafy_url;

                            if (btnA) {
                                btnA.disabled = status === 'Assinado';
                                btnA.innerHTML = status === 'Assinado' ? '<i class="ph ph-pen-nib"></i> Assinado' : '<i class="ph ph-pen-nib"></i> Assinar p/ Assinafy';
                            }
                            if (badgeA) {
                                badgeA.innerText = status !== 'Nenhum' ? status.toUpperCase() : '';
                                badgeA.className = 'assinafy-status-badge ' + status.toLowerCase();
                            }

                            if (url) {
                                if (!btnW) {
                                    btnW = document.createElement('button');
                                    btnW.className = 'btn btn-sm btn-whatsapp-integrated';
                                    btnW.style.cssText = 'background:#25D366; color:white; padding: 0.3rem 0.5rem;';

                                };

                            }
                        }
